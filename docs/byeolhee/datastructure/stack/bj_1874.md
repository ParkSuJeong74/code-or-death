# [스택 수열](https://www.acmicpc.net/problem/1874)

```python
import sys
def input():
  return sys.stdin.readline().rstrip()

# 첫 번째 입력: n
n = int(input())
stack = []
answer = []
flag = 0
cur = 1

for i in range(n):
  num = int(input())
  while cur <= num:
    # 원하는 수를 pop하기 위해 push
    stack.append(cur)
    answer.append('+')
    cur += 1

  if stack[-1] == num:
    stack.pop()
    answer.append('-')
  else:
    print('NO')
    flag = 1
    break

if flag == 0:
  for i in answer:
    print(i)
```
